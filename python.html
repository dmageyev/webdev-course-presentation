🔁 ЦИКЛИ — АВТОМАТИЧНЕ ПОВТОРЕННЯ

Замість того щоб робити 1000 дій вручну — один цикл!

# Цикл for — перебираємо список
students = ["Іван", "Марія", "Олег", "Анна"]

for name in students:
    print(f"Привіт, {name}!")

# range() — числовий діапазон
total = 0
for i in range(1, 11):   # від 1 до 10
    total += i
print(f"Сума 1..10 = {total}")

# ---
# Цикл while — поки умова True

while count > 0:
    print(f"Залишилось спроб: {count}")
    count -= 1
print("Доступ закрито")

🛠️ ФУНКЦІЇ — ГОТОВІ БЛОКИ КОДУ

Функція — іменований блок, який вирішує одне конкретне завдання

def greet(name):
    """Повертає привітальний рядок."""
    return f"Привіт, {name}! 👋"

def calculate_discount(price, percent):
    """Рахує ціну зі знижкою."""
    saving = price * percent / 100
    return round(price - saving, 2)

def letter_grade(score):
    """Конвертує бал у літерну оцінку."""
    if   score >= 90: return "A"
    elif score >= 75: return "B"
    elif score >= 60: return "C"
    else:             return "F"

# Виклик функцій
print(greet("Олена"))
print(calculate_discount(1500, 20))

scores = [88, 72, 95, 54, 61]
for s in scores:
    print(f"{s} → {letter_grade(s)}")

import smtplib, pandas as pd

# Читаємо список студентів
df = pd.read_excel("students.xlsx")

server = smtplib.SMTP("smtp.gmail.com", 587)
server.starttls()
server.login("you@gmail.com", "pass")

for _, row in df.iterrows():
    msg = f"""
    Привіт, {row['name']}!
    Ваша оцінка: {row['grade']}
    """
    server.sendmail(
        "you@gmail.com",
        row['email'],
        msg
    )

server.quit()
print("Розсилку завершено ✅")

Завдання: щодня автоматично перевіряти ціни на товари у 5 інтернет-магазинах

import requests
from bs4 import BeautifulSoup
import pandas as pd
from datetime import date

urls = {
    "Shop A": "https://shop-a.ua/product/123",
    "Shop B": "https://shop-b.ua/product/456",
}

results = []

for shop, url in urls.items():
    response = requests.get(url)
    soup = BeautifulSoup(
        response.text, "html.parser"
    )
    price = soup.find(
        "span", class_="price"
    ).text.strip()

    results.append({
        "Магазин": shop,
        "Ціна": price,
        "Дата": date.today()
    })

df = pd.DataFrame(results)
df.to_excel("prices.xlsx", index=False)
print(df)
